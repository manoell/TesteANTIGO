        // ==== Funcionalidades de Inicialização ====

        async function initialize() {

            try {

                updateStatus('Inicializando...', 'connecting');


                // Verificar suporte a getUserMedia

                if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {

                    throw new Error('Seu navegador não suporta acesso à câmera (getUserMedia)');

                }


                // Listar dispositivos de vídeo disponíveis

                const devices = await navigator.mediaDevices.enumerateDevices();

                const videoDevices = devices.filter(device => device.kind === 'videoinput');


                // Adicionar opções ao select

                videoSource.innerHTML = '';

                videoDevices.forEach(device => {

                    const option = document.createElement('option');

                    option.value = device.deviceId;

                    option.text = device.label || `Câmera ${videoSource.length + 1}`;

                    videoSource.appendChild(option);

                });


                // Se não houver rótulos, solicitar permissão de câmera

                if (videoDevices.length > 0 && !videoDevices[0].label) {

                    await getLocalMediaStream();

                    // Listar novamente para obter rótulos

                    await initialize();

                    return;

                }


                if (videoDevices.length === 0) {

                    updateStatus('Nenhuma câmera encontrada', 'offline');

                } else {

                    updateStatus('Pronto para transmitir', 'offline');

                }


                // Configurar IDs de câmera preferenciais para melhor qualidade

                const preferredCameras = videoDevices.filter(device => 

                    device.label.toLowerCase().includes('hd') ||

                    device.label.toLowerCase().includes('4k') ||

                    device.label.toLowerCase().includes('pro')

                );


                if (preferredCameras.length > 0) {

                    videoSource.value = preferredCameras[0].deviceId;

                }


            } catch (error) {

                console.error('Erro ao inicializar:', error);

                updateStatus(`Erro: ${error.message}`, 'offline');

            }

        }


        // Obter stream local com qualidade configurada

        async function getLocalMediaStream() {

            try {

                if (localStream) {

                    localStream.getTracks().forEach(track => track.stop());

                }


                // Obter qualidade selecionada

                const quality = videoQualities[videoQuality.value];

                const fps = parseInt(frameRate.value);


                console.log(`Configurando vídeo: ${quality.width}x${quality.height} a ${fps}fps`);


                const constraints = {

                    audio: false,  // Não precisamos de áudio para VCAM

                    video: {

                        deviceId: videoSource.value ? { exact: videoSource.value } : undefined,

                        width: { ideal: quality.width },

                        height: { ideal: quality.height },

                        frameRate: { ideal: fps }

                    }

                };


                localStream = await navigator.mediaDevices.getUserMedia(constraints);


                // Log das configurações reais obtidas

                const videoTrack = localStream.getVideoTracks()[0];

                if (videoTrack) {

                    const settings = videoTrack.getSettings();

                    console.log('Configurações reais do vídeo:', settings);

                    

                    // Atualizar badges de qualidade

                    resolutionBadge.textContent = `${settings.width}x${settings.height}`;

                    fpsBadge.textContent = `${settings.frameRate || 'N/A'} FPS`;

                    codecBadge.textContent = videoTrack.contentHint || 'H.264';

                    

                    videoStats.innerHTML = `Fonte: ${videoTrack.label}<br>` +

                                      `Resolução: ${settings.width}x${settings.height}<br>` +

                                      `FPS: ${settings.frameRate || 'desconhecido'}<br>` +

                                      `Aspect Ratio: ${(settings.width/settings.height).toFixed(2)}`;

                    statsPanel.classList.remove('hidden');

                }


                localVideo.srcObject = localStream;

                return localStream;

            } catch (error) {

                console.error('Erro ao acessar câmera:', error);

                updateStatus(`Erro ao acessar câmera: ${error.message}`, 'offline');

                throw error;

            }

        }


        // ==== WebRTC e WebSocket ====

        // Configurar conexão WebRTC

        function setupPeerConnection() {

            // Criar nova conexão

            peerConnection = new RTCPeerConnection(rtcConfig);


            // Adicionar stream de vídeo

            localStream.getTracks().forEach(track => {

                console.log(`Adicionando track: ${track.kind}`);

                peerConnection.addTrack(track, localStream);

            });


            // Configurar eventos

            peerConnection.onicecandidate = event => {

                if (event.candidate) {

                    sendMessage({

                        type: 'ice-candidate',

                        candidate: event.candidate.candidate,

                        sdpMid: event.candidate.sdpMid,

                        sdpMLineIndex: event.candidate.sdpMLineIndex,

                        roomId: roomIdInput.value

                    });

                }

            };


            peerConnection.oniceconnectionstatechange = () => {

                console.log('ICE State:', peerConnection.iceConnectionState);

                connectionStats.innerHTML = `ICE State: ${peerConnection.iceConnectionState}`;


                // Atualizar qualidade da conexão com base no estado ICE

                updateConnectionQualityIndicators(peerConnection.iceConnectionState);


                if (peerConnection.iceConnectionState === 'connected' ||

                    peerConnection.iceConnectionState === 'completed') {

                    updateStatus('Conectado ao dispositivo iOS', true);

                    isConnectionActive = true;

                } else if (peerConnection.iceConnectionState === 'disconnected' ||

                          peerConnection.iceConnectionState === 'failed' ||

                          peerConnection.iceConnectionState === 'closed') {

                    updateStatus('Dispositivo desconectado', false);

                    isConnectionActive = false;

                    

                    // Tentar reconexão automática (apenas para desconexões)

                    if (peerConnection.iceConnectionState === 'disconnected' && !reconnectTimeout) {

                        reconnectTimeout = setTimeout(() => {

                            if (isConnectionActive === false && !stopButton.disabled) {

                                console.log('Tentando reconexão automática...');

                                createAndSendOffer();

                                reconnectTimeout = null;

                            }

                        }, 5000);

                    }

                }

            };


            // Configurar envio periódico de estatísticas para depuração

            startStatsMonitor();

        }


        // Atualizar indicadores de qualidade de conexão

        function updateConnectionQualityIndicators(state) {

            // Resetar indicadores

            qualityIndicators.forEach(ind => ind.classList.remove('active'));

            

            // Ativar indicadores com base no estado

            switch(state) {

                case 'new':

                case 'checking':

                    qualityIndicators[0].classList.add('active');

                    break;

                case 'connected':

                    qualityIndicators[0].classList.add('active');

                    qualityIndicators[1].classList.add('active');

                    qualityIndicators[2].classList.add('active');

                    break;

                case 'completed':

                    qualityIndicators.forEach(ind => ind.classList.add('active'));

                    break;

                case 'disconnected':

                    qualityIndicators[0].classList.add('active');

                    break;

                case 'failed':

                case 'closed':

                    // Todos desativados

                    break;

            }

        }


        // Iniciar monitoramento de estatísticas

        function startStatsMonitor() {

            if (statsInterval) {

                clearInterval(statsInterval);

            }


            statsInterval = setInterval(async () => {

                if (!peerConnection) return;


                const stats = await peerConnection.getStats();

                let outboundRtpStats;

                let currentFrameCount = 0;


                stats.forEach(stat => {

                    if (stat.type === 'outbound-rtp' && stat.kind === 'video') {

                        outboundRtpStats = stat;

                        currentFrameCount = stat.framesSent || 0;

                    }

                });


                if (outboundRtpStats) {

                    const { bytesSent, packetsSent, framesSent, framesPerSecond: reportedFPS } = outboundRtpStats;

                    

                    // Calcular FPS se não for relatado diretamente

                    if (!reportedFPS && lastFrameCount > 0) {

                        framesPerSecond = currentFrameCount - lastFrameCount;

                    } else {

                        framesPerSecond = reportedFPS || 0;

                    }

                    lastFrameCount = currentFrameCount;

                    

                    // Atualizar a badge de FPS com o valor atual

                    fpsBadge.textContent = `${Math.round(framesPerSecond)} FPS`;


                    connectionStats.innerHTML = `ICE State: ${peerConnection.iceConnectionState}<br>` +

                                               `Dados enviados: ${formatBytes(bytesSent)}<br>` +

                                               `Pacotes enviados: ${packetsSent}<br>` +

                                               `Frames enviados: ${framesSent}<br>` +

                                               `FPS atual: ${framesPerSecond || 'desconhecido'}`;

                                               

                    // Ajustar qualidade com base em estatísticas

                    updateConnectionQuality(bytesSent, framesPerSecond);

                }

            }, 1000);

        }


        // Ajustar indicadores de qualidade baseado em métricas

        function updateConnectionQuality(bytesSent, fps) {

            let qualityLevel = 0;

            

            // Verifica taxa de bits (bitrate)

            const kbitsPerSecond = bytesSent * 8 / 1000; // Convertendo bytes para kilobits

            

            if (kbitsPerSecond > 5000) qualityLevel++; // > 5 Mbps

            if (kbitsPerSecond > 15000) qualityLevel++; // > 15 Mbps

            if (kbitsPerSecond > 30000) qualityLevel++; // > 30 Mbps

            if (kbitsPerSecond > 45000) qualityLevel++; // > 45 Mbps

            

            // Ajuste baseado em FPS

            if (fps < 15 && qualityLevel > 0) qualityLevel--;

            

            // Atualizar indicadores

            qualityIndicators.forEach((ind, index) => {

                ind.classList.toggle('active', index < qualityLevel);

            });

        }


        // Formatar bytes para exibição

        function formatBytes(bytes, decimals = 2) {

            if (bytes === 0) return '0 Bytes';


            const k = 1024;

            const sizes = ['Bytes', 'KB', 'MB', 'GB'];

            const i = Math.floor(Math.log(bytes) / Math.log(k));


            return parseFloat((bytes / Math.pow(k, i)).toFixed(decimals)) + ' ' + sizes[i];

        }


        // Conectar ao servidor WebSocket

        function connectWebSocket() {

            const port = window.location.port || '8080';

            const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';

            const urlString = `${protocol}//localhost:${port}`;


            console.log(`Conectando WebSocket a ${urlString}`);

            updateStatus('Conectando ao servidor...', 'connecting');


            // Fechar conexão existente

            if (ws) {

                ws.close();

                ws = null;

            }


            ws = new WebSocket(urlString);


            ws.onopen = () => {

                console.log('WebSocket conectado');

                sendMessage({

                    type: 'join',

                    roomId: roomIdInput.value

                });

                updateStatus('Conectado ao servidor', 'connecting');

                

                // Iniciar envio periódico de keepalive

                if (connectionCheckInterval) clearInterval(connectionCheckInterval);

                connectionCheckInterval = setInterval(() => {

                    if (ws && ws.readyState === WebSocket.OPEN) {

                        ws.send(JSON.stringify({ type: 'keepalive' }));

                    }

                }, 15000);

            };


            ws.onclose = () => {

                console.log('WebSocket desconectado');

                updateStatus('Desconectado do servidor', false);

                

                clearInterval(connectionCheckInterval);


                // Tentar reconectar após 3 segundos apenas se a conexão foi perdida inesperadamente

                if (startButton.disabled && !stopButton.disabled) {

                    setTimeout(() => {

                        connectWebSocket();

                    }, 3000);

                }

            };


            ws.onerror = (error) => {

                console.error('WebSocket erro:', error);

                updateStatus('Erro de conexão', 'offline');

            };


            ws.onmessage = (event) => {

                const message = JSON.parse(event.data);

                handleMessage(message);

            };

        }


        // Enviar mensagem para o servidor

        function sendMessage(message) {

            if (ws && ws.readyState === WebSocket.OPEN) {

                message.timestamp = Date.now();

                ws.send(JSON.stringify(message));

            }

        }


        // Processar mensagens recebidas

        async function handleMessage(message) {

            console.log('Mensagem recebida:', message.type);


            switch (message.type) {

                case 'user-joined':

                    console.log('Dispositivo iOS conectado');

                    // Criar oferta quando o iOS se conectar

                    await createAndSendOffer();

                    break;

                    

                case 'answer':

                    // Recebendo resposta do iOS

                    await peerConnection.setRemoteDescription(new RTCSessionDescription({

                        type: 'answer',

                        sdp: message.sdp

                    }));

                    break;

                    

                case 'ice-candidate':

                    // Recebendo candidato ICE

                    if (peerConnection) {

                        try {

                            await peerConnection.addIceCandidate(new RTCIceCandidate({

                                candidate: message.candidate,

                                sdpMid: message.sdpMid,

                                sdpMLineIndex: message.sdpMLineIndex

                            }));

                        } catch (e) {

                            console.error('Erro ao adicionar candidato ICE:', e);

                        }

                    }

                    break;

                    

                case 'connection-config':

                    // Recebendo configurações otimizadas do servidor

                    console.log('Configurações recebidas do servidor:', message);

                    break;

                    

                case 'error':

                    console.error('Erro recebido do servidor:', message.message);

                    updateStatus(`Erro: ${message.message}`, 'offline');

                    break;

                    

                case 'keepalive-ack':

                    // Simplesmente ignorar, é apenas para manter a conexão viva

                    break;

                    

                case 'server-info':

                    console.log('Informações do servidor:', message);

                    break;

            }

        }


        // Criar e enviar oferta WebRTC

        async function createAndSendOffer() {

            try {

                // Configurar para alta qualidade de vídeo

                const offerOptions = {

                    offerToReceiveAudio: false,

                    offerToReceiveVideo: false

                };


                const offer = await peerConnection.createOffer(offerOptions);


                // Modificar SDP para forçar parâmetros de alta qualidade

                let sdp = offer.sdp;


                // Forçar H.264 quando disponível e otimizar seus parâmetros

                if (sdp.includes('H264')) {

                    console.log('H264 disponível, otimizando para alta qualidade');


                    // Melhorar profile do H264 para alta resolução

                    sdp = sdp.replace(/profile-level-id=[0-9a-fA-F]+/g, 'profile-level-id=640032');


                    // Adicionar parâmetros adicionais para H.264 de alta qualidade

                    if (!sdp.includes('level-asymmetry-allowed')) {

                        sdp = sdp.replace(/a=fmtp:(\d+) (packetization-mode=1(;?))/g, 

                            'a=fmtp:$1 packetization-mode=1;level-asymmetry-allowed=1$3');

                    }


                    // Aumentar bitrate para vídeo 4K

                    sdp = setMediaBitrate(sdp, 'video', 50000); // 50Mbps para WiFi 5GHz

                }


                // Aplicar a SDP modificada

                const modifiedOffer = new RTCSessionDescription({

                    type: 'offer',

                    sdp: sdp

                });


                await peerConnection.setLocalDescription(modifiedOffer);


                sendMessage({

                    type: 'offer',

                    sdp: peerConnection.localDescription.sdp,

                    roomId: roomIdInput.value

                });

            } catch (error) {

                console.error('Erro ao criar oferta:', error);

                updateStatus(`Erro ao criar oferta WebRTC: ${error.message}`, 'offline');

            }

        }


        // Helper para configurar bitrate específico na SDP

        function setMediaBitrate(sdp, media, bitrate) {

            const lines = sdp.split('\n');

            let line = -1;


            // Encontrar a seção m=video

            for (let i = 0; i < lines.length; i++) {

                if (lines[i].startsWith('m=' + media)) {

                    line = i;

                    break;

                }

            }


            if (line === -1) {

                return sdp;

            }


            // Verificar se já existe uma linha b=AS:

            let found = false;

            for (let i = line; i < lines.length; i++) {

                if (lines[i].startsWith('m=')) break; // Próxima seção m=


                if (lines[i].startsWith('b=AS:')) {

                    lines[i] = 'b=AS:' + bitrate;

                    found = true;

                    break;

                }

            }


            // Se não encontrou, adicionar após a seção m=

            if (!found) {

                const newLines = lines.slice(0, line + 1);

                newLines.push('b=AS:' + bitrate);

                newLines.push('b=TIAS:' + (bitrate * 1000));

                newLines.push(...lines.slice(line + 1));

                return newLines.join('\n');

            }


            // Caso contrário, retornar o SDP modificado

            return lines.join('\n');

        }


        // Atualizar status na interface

        function updateStatus(message, state) {

            statusDiv.textContent = message;

            

            // Remover todas as classes de status

            statusDiv.classList.remove('status-offline', 'status-online', 'status-connecting');

            

            // Adicionar a classe apropriada

            if (state === true || state === 'online') {

                statusDiv.classList.add('status-online');

            } else if (state === 'connecting') {

                statusDiv.classList.add('status-connecting');

            } else {

                statusDiv.classList.add('status-offline');

            }

        }

        // Parar transmissão

        function stopStreaming() {

            startButton.disabled = false;

            stopButton.disabled = true;

            roomIdInput.disabled = false;

            videoSource.disabled = false;

            videoQuality.disabled = false;

            frameRate.disabled = false;

            

            isConnectionActive = false;

            

            // Cancelar qualquer tentativa de reconexão

            if (reconnectTimeout) {

                clearTimeout(reconnectTimeout);

                reconnectTimeout = null;

            }


            // Enviar mensagem de despedida

            if (ws && ws.readyState === WebSocket.OPEN) {

                sendMessage({

                    type: 'bye',

                    roomId: roomIdInput.value

                });

            }


            // Parar monitoramento de estatísticas

            if (statsInterval) {

                clearInterval(statsInterval);

                statsInterval = null;

            }

            

            if (connectionCheckInterval) {

                clearInterval(connectionCheckInterval);

                connectionCheckInterval = null;

            }


            // Fechar conexão WebRTC

            if (peerConnection) {

                peerConnection.close();

                peerConnection = null;

            }


            // Fechar WebSocket

            if (ws) {

                ws.close();

                ws = null;

            }


            // Parar streams

            if (localStream) {

                localStream.getTracks().forEach(track => track.stop());

                localStream = null;

                localVideo.srcObject = null;

            }


            // Esconder painel de estatísticas

            statsPanel.classList.add('hidden');

            

            // Resetar badges

            resolutionBadge.textContent = "—";

            fpsBadge.textContent = "—";

            codecBadge.textContent = "—";

            

            // Resetar indicadores de qualidade

            qualityIndicators.forEach(ind => ind.classList.remove('active'));


            updateStatus('Desconectado', 'offline');

        }

        startButton.addEventListener('click', startStreaming);

        stopButton.addEventListener('click', stopStreaming);

        videoQuality.addEventListener('change', async () => {

                    console.error("Erro ao atualizar stream:", error);

                }

            }

        });


        // Verificar perda de conexão automaticamente

        function setupConnectionCheck() {

            // A verificação principal agora é feita pelo keepalive no WebSocket

            setInterval(() => {

                if (isConnectionActive && peerConnection && 

                    (peerConnection.iceConnectionState === 'disconnected' || 

                     peerConnection.iceConnectionState === 'failed')) {

                    

                    console.log('Conexão perdida, verificando estado do WebSocket...');

                    

                    // Se WebSocket estiver ok mas ICE não, tentar recriar apenas o peer connection

                    if (ws && ws.readyState === WebSocket.OPEN) {

                        console.log('WebSocket ativo, tentando recriar peer connection...');

                        

                        // Fechar conexão anterior e criar nova

                        if (peerConnection) {

                            peerConnection.close();

                        }

                        

                        // Recriar peer connection e enviar nova oferta

                        setupPeerConnection();

                        createAndSendOffer();

                    }

                }

            }, 10000);

        }
